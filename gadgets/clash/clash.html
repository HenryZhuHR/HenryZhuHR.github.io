export sys_proxy_lip=192.168.1.115
export sys_proxy_port=7890
# 开启系统代理
proxy_on() {
	export http_proxy=$sys_proxy_lip:$sys_proxy_port
	export https_proxy=$sys_proxy_lip:$sys_proxy_port
	export no_proxy=127.0.0.1,localhost
	export HTTP_PROXY=$sys_proxy_lip:$sys_proxy_port
	export HTTPS_PROXY=$sys_proxy_lip:$sys_proxy_port
 	export NO_PROXY=127.0.0.1,localhost
	echo -e "\033[32m[√] 已开启代理\033[0m"
}
# 关闭系统代理
proxy_off(){
	unset http_proxy
	unset https_proxy
	unset no_proxy
  	unset HTTP_PROXY
	unset HTTPS_PROXY
	unset NO_PROXY
	echo -e "\033[31m[×] 已关闭代理\033[0m"
}
