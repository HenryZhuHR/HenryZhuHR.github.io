自动同步fork的上游分支 ​

yml
# .github/workflows/sync-fork.yml
name: Sync Fork

on:
  schedule:
    - cron: '0 0 * * *' # every day
  workflow_dispatch: # on button click

jobs:
  sync:

    runs-on: ubuntu-latest

    steps:
      - uses: tgymnich/fork-sync@v1.9.0
        # https://github.com/tgymnich/fork-sync
        with:
          owner: meta-llama # 上有分支仓库所述者
          base: main
          head: main
